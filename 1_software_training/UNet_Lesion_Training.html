class MriLesionDatasetTF(tf.keras.utils.Sequence):
    def __init__(self, root_dir, batch_size=4, patch_size=144):
        self.root = root_dir
        self.batch = batch_size
        self.patch = patch_size

        cancer = os.listdir(f"{root_dir}/images/cancer")
        non_cancer = os.listdir(f"{root_dir}/images/non_cancer")
        non_cancer = non_cancer[:len(cancer)//2]

        self.samples = (
            [("cancer", p) for p in cancer] +
            [("non_cancer", p) for p in non_cancer]
        )

    def __len__(self):
        return int(np.ceil(len(self.samples) / self.batch))

    # ----------------------------
    # image loading
    # ----------------------------
    def load_img(self, path):
        img = cv2.imread(path, cv2.IMREAD_GRAYSCALE).astype(np.float32)
        img = (img - img.mean()) / (img.std() + 1e-5)
        return img

    def load_mask(self, path):
        mask = cv2.imread(path, cv2.IMREAD_GRAYSCALE)
        return (mask > 0).astype(np.float32)

    # ----------------------------
    # EXACT crop logic (PyTorch match)
    # ----------------------------
    def crop_patch(self, img, mask):
        patch = self.patch
        _, H, W = img.shape
        mask2d = mask[0]

        ys, xs = np.where(mask2d > 0)
        if len(xs) == 0:
            y, x = H // 2, W // 2
        else:
            y, x = int(ys.mean()), int(xs.mean())

        s = patch // 2
        y1, y2 = y - s, y + s
        x1, x2 = x - s, x + s

        pad_t = max(0, -y1)
        pad_l = max(0, -x1)
        pad_b = max(0, y2 - H)
        pad_r = max(0, x2 - W)

        img  = np.pad(img,  ((0,0),(pad_t,pad_b),(pad_l,pad_r)))
        mask = np.pad(mask, ((0,0),(pad_t,pad_b),(pad_l,pad_r)))

        y1 += pad_t; y2 += pad_t
        x1 += pad_l; x2 += pad_l

        return img[:, y1:y2, x1:x2], mask[:, y1:y2, x1:x2]

    # ----------------------------
    # REQUIRED by Sequence
    # ----------------------------
    def __getitem__(self, idx):
        batch = self.samples[idx*self.batch:(idx+1)*self.batch]

        images, masks = [], []

        for label, pid in batch:
            base = f"{self.root}/images/{label}/{pid}/{pid}"

            t2w = self.load_img(base + "_t2w.png")
            hbv = self.load_img(base + "_hbv.png")
            adc = self.load_img(base + "_adc.png")

            img = np.stack([t2w, hbv, adc], axis=0)

            mask = self.load_mask(
                f"{self.root}/masks/{label}/{pid}.png"
            )[None]

            img, mask = self.crop_patch(img, mask)

            # convert CHW → HWC
            images.append(np.transpose(img, (1,2,0)))
            masks.append(np.transpose(mask, (1,2,0)))

        return np.array(images), np.array(masks)

def DoubleConv(x, filters):
    x = tf.keras.layers.Conv2D(filters, 3, padding="same", use_bias=False)(x)
    x = tf.keras.layers.BatchNormalization()(x)
    x = tf.keras.layers.ReLU()(x)
    x = tf.keras.layers.Conv2D(filters, 3, padding="same", use_bias=False)(x)
    x = tf.keras.layers.BatchNormalization()(x)
    x = tf.keras.layers.ReLU()(x)
    return x

def LesionUNet(input_shape=(144,144,3)):
    inputs = tf.keras.Input(input_shape)

    e1 = DoubleConv(inputs, 32)
    e2 = DoubleConv(tf.keras.layers.MaxPool2D()(e1), 64)
    e3 = DoubleConv(tf.keras.layers.MaxPool2D()(e2), 128)
    e4 = DoubleConv(tf.keras.layers.MaxPool2D()(e3), 256)

    b  = DoubleConv(tf.keras.layers.MaxPool2D()(e4), 512)

    d4 = tf.keras.layers.Concatenate()([
        e4, tf.keras.layers.UpSampling2D()(b)
    ])
    d4 = DoubleConv(d4, 256)

    d3 = tf.keras.layers.Concatenate()([
        e3, tf.keras.layers.UpSampling2D()(d4)
    ])
    d3 = DoubleConv(d3, 128)

    d2 = tf.keras.layers.Concatenate()([
        e2, tf.keras.layers.UpSampling2D()(d3)
    ])
    d2 = DoubleConv(d2, 64)

    d1 = tf.keras.layers.Concatenate()([
        e1, tf.keras.layers.UpSampling2D()(d2)
    ])
    d1 = DoubleConv(d1, 32)

    outputs = tf.keras.layers.Conv2D(1, 1)(d1)

    return tf.keras.Model(inputs, outputs)

def tf_run_inference_and_visualize(model, dataset, root_dir, idx=0):
    """
    TF inference visualization (EXACT PyTorch behavior)
    """

    # Dataset already returns batch of size 1
    image, mask = dataset[idx]      # image: (1,144,144,3)
    image = tf.squeeze(image, axis=0)   # (144,144,3)
    mask  = tf.squeeze(mask, axis=0)    # (144,144,1)

    label, pid = dataset.samples[idx]
    base = f"{root_dir}/images/{label}/{pid}/{pid}"

    # Original image
    t2w_org = cv2.imread(base + "_t2w.png", 0)
    gt_org  = cv2.imread(f"{root_dir}/masks/{label}/{pid}.png", 0) > 0

    # ---- TF inference ----
    image_tf = tf.expand_dims(image, axis=0)   # (1,144,144,3)
    logits   = model(image_tf, training=False)

    prob = tf.nn.sigmoid(logits)[0, :, :, 0].numpy()
    pred = prob > 0.5

    # ---- Visualization ----
    fig, ax = plt.subplots(1, 4, figsize=(18, 5))

    ax[0].imshow(t2w_org, cmap="gray")
    ax[0].imshow(gt_org, cmap="Reds", alpha=0.4)
    ax[0].set_title("Original T2W + GT")

    ax[1].imshow(image[:, :, 0], cmap="gray")
    ax[1].imshow(mask[:, :, 0], cmap="Reds", alpha=0.4)
    ax[1].set_title("Cropped T2W + GT")

    ax[2].imshow(image[:, :, 0], cmap="gray")
    ax[2].imshow(pred, cmap="Reds", alpha=0.4)
    ax[2].set_title("Cropped T2W + Prediction")

    im = ax[3].imshow(prob, cmap="jet")
    ax[3].set_title("Probability Map")
    plt.colorbar(im, ax=ax[3])

    for a in ax:
        a.axis("off")

    plt.tight_layout()
    plt.show()

def iou_score_tf(y_true, y_pred, eps=1e-6):
    y_true = tf.cast(y_true > 0.5, tf.float32)
    y_pred = tf.cast(y_pred > 0.5, tf.float32)

    intersection = tf.reduce_sum(y_true * y_pred)
    union = tf.reduce_sum(y_true) + tf.reduce_sum(y_pred) - intersection

    return (intersection + eps) / (union + eps)


def precision_tf(y_true, y_pred, eps=1e-6):
    y_true = tf.cast(y_true > 0.5, tf.float32)
    y_pred = tf.cast(y_pred > 0.5, tf.float32)

    tp = tf.reduce_sum(y_true * y_pred)
    fp = tf.reduce_sum((1 - y_true) * y_pred)

    return (tp + eps) / (tp + fp + eps)


def recall_tf(y_true, y_pred, eps=1e-6):
    y_true = tf.cast(y_true > 0.5, tf.float32)
    y_pred = tf.cast(y_pred > 0.5, tf.float32)

    tp = tf.reduce_sum(y_true * y_pred)
    fn = tf.reduce_sum(y_true * (1 - y_pred))

    return (tp + eps) / (tp + fn + eps)


def accuracy_tf(y_true, y_pred):
    y_true = tf.cast(y_true > 0.5, tf.float32)
    y_pred = tf.cast(y_pred > 0.5, tf.float32)

    correct = tf.reduce_sum(tf.cast(tf.equal(y_true, y_pred), tf.float32))
    total = tf.cast(tf.size(y_true), tf.float32)

    return correct / total

def evaluate_tf_model(model, dataset):
    dice_list = []
    iou_list = []
    precision_list = []
    recall_list = []
    accuracy_list = []

    for images, masks in dataset:
        # images: (B,144,144,3)
        # masks : (B,144,144,1)

        logits = model(images, training=False)
        probs = tf.nn.sigmoid(logits)

        dice_list.append(dice_score_tf(masks, probs).numpy())
        iou_list.append(iou_score_tf(masks, probs).numpy())
        precision_list.append(precision_tf(masks, probs).numpy())
        recall_list.append(recall_tf(masks, probs).numpy())
        accuracy_list.append(accuracy_tf(masks, probs).numpy())

    return {
        "dice": np.mean(dice_list),
        "iou": np.mean(iou_list),
        "precision": np.mean(precision_list),
        "recall": np.mean(recall_list),
        "accuracy": np.mean(accuracy_list),
    }
